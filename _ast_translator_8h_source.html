61  struct Location {
62  int level; // loop level
63  int element; // element in a tuple
64  std::string name; // name of associated attribute
65 
66  bool operator==(const Location& loc) const {
67  return level == loc.level && element == loc.element;
68  }
69 
70  bool operator!=(const Location& loc) const {
71  return !(*this == loc);
72  }
73 
74  bool operator<(const Location& loc) const {
75  return level < loc.level || (level == loc.level && element < loc.element);
76  }
77

187  auto pos = record_unpacks.find(&init);

197  void setAggregatorLocation(const AstAggregator& agg, const Location& loc) {
198  aggregator_locations.push_back(std::make_pair(&agg, loc));
199  }
200 
201  const Location& getAggregatorLocation(const AstAggregator& agg) const {
202  // search list
203  for (const auto& cur : aggregator_locations) {
204  if (*cur.first == agg) {
205  return cur.second;
206  }
207  }
208 
209  // fail
210  std::cout << "Lookup of " << &agg << " = " << agg << " failed\n";
211  assert(false && "Requested aggregation operation is not processed!");
212 
213  const static Location fail = Location();
214  return fail;
215  }

217  // -- others --
218 
219  bool isSomethingDefinedOn(int level) const {
220  // check for variable definitions
221  for (const auto& cur : var_references) {
222  if (cur.second.begin()->level == level) {
223  return true;
224  }
225  }
226  // check for record definitions
227  for (const auto& cur : record_definitions) {
228  if (cur.second.level == level) {
229  return true;
230  }
231  }
232  // nothing defined on this level
233  return false;
234  }
